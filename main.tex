\section{Motivation}
\subsection{The Climate Free‑Rider Problem}

\begin{frame}{Humanity's Greatest Threat}

    \begin{itemize}
    \setlength{\itemsep}{8pt}
        \item \textbf{Global Warming}\footnote{'Global warming' is used to avoid confusion with short-term or seasonal weather changes, or ozone depletion \cite{Lorenzoni2006}.}
        
        \begin{itemize}
            \item \textbf{10–25\%} Global GDP losses by 2100 [Undiminished] (SSP3-7.0 scenario)
            \item \textbf{0.5–5\%} Global GDP losses by 2100 [1.5–2°C] (Paris Agreement Target)
            \\ Stated by the \cite{ipcc2023}.
        \end{itemize}

        \item \textbf{Climate protection makes economic sense:} \\ 
        For every 1€ invested in climate mitigation, 4€ in avoided climate damages
    \end{itemize}

    \medskip
    
    \begin{alertblock}{A classic problem of free riding}
    The benefits of climate protection are globally distributed, but protecting the climate comes at a personal cost. How can we fight global warming?
    \end{alertblock}
    
\end{frame}

\begin{frame}{Research Questions}

    \textcite{andre2024} provides the first comprehensive picture of the US climate context using a representative sample and an incentivised donation decision.
    
    \medskip
    
    \begin{enumerate}
    \setlength{\itemsep}{8pt}
        \item What factors drive individual willingness to fight Global Warming? \textbf{Behavioural Determinants}
        \item Do misperceptions of social norms influence climate action, and if so, how? \textbf{Social Norms}
        \item Can correcting misperceived norms increase support for climate policies? \textbf{Intervention Effect}
    \end{enumerate}

\end{frame}

\section{Study Design}

\begin{frame}{How to measure Individuals' Willingness to fight Global Warming}
    \begin{itemize}
    \setlength{\itemsep}{8pt}
        \item Quota-Based sampling approach of 8,000 participants \footnote{Recrutement done by \href{https://business.pureprofile.com/who-we-are/}{Pureprofile}}
        
        \begin{itemize}
            \item \textbf{Representing} the adult US population in terms of key socio-demographic variables, allowing for \textbf{mitigation of concerns about selection}.
            \item Two separate collection waves with $n=2000$ and $n=6000$.
            \item Respondents could only participate in one of the two waves.
        \end{itemize}

        \item Collect detailed background information of individuals' moral values, preferences and beliefs about social norms.
        \item Study the \textbf{impact of the information treatment} and examine if the treatment effects of information are heterogeneous across all kinds of groups.
        
    \end{itemize}
    
\end{frame}

\subsection{Wave 1: Descriptive Analysis (n = 2,000, March 2021)}

    \begin{itemize}
        \item Results are quantitative \& interpersonally comparable.
        \item Probabilistic incentive scheme: Decisions of 25 randomised participants are implemented.
    \end{itemize}
    
\end{frame}

\begin{frame}{Wave 1: Measuring Behaviour Determinants}
\framesubtitle{Descriptive Analysis (n = 2,000, March 2021)}
    \begin{itemize}
    \setlength{\itemsep}{8pt}
        \item Baseline behaviour and endorsements of norm \footnote{The solution means are used as the Information treatments in Wave 2 \medskip}
        
        \begin{itemize}
        \setlength{\itemsep}{4pt}
            \item Behaviour: 'Do you try to ﬁght global warming?' [Yes/No] 
            \item Norms: 'Do you think that people in the United States should try to ﬁght global warming?' [Yes/No]
        \end{itemize}
        
        \item Perceived social norms
        
        \begin{itemize}
        \setlength{\itemsep}{4pt}
            \item Perceived behaviour: 'Out of 100 people, how many stated that they try to ﬁght global warming?' [0-100]
            \item Perceived norms: 'Out of 100 people, how many stated that they think that people in the United States should try to ﬁght global warming?' [0-100]
        \end{itemize}

        \item Additional Measurements: Policy support, Political engagement, Posterior beliefs about social norms, Climate change scepticism, Background characteristics.
        
    \end{itemize}
    
\end{frame}

\section{Conclusion \& Implications}

\begin{frame}{Conclusion \& Implications}

    \begin{block}{Conclusion}
    Norms matter $\rightarrow$ They’re misperceived $\rightarrow$ Correcting them works $\rightarrow$ Especially for skeptics $\rightarrow$ Reducing polarization
    \end{block}

    \medskip

    \begin{itemize}
        \item Behavioural Determinants: Social norms and economic preferences predict Global Warming behaviour
        \item Misperceptions of Social Norms: Americans systematically underestimate the share of Americans trying to fight global warming
    \end{itemize}

    \medskip

    \begin{alertblock}{Treatment Effects}
        Statistically significant treatment effects on Donations, posterior Behaviour, and Norms belief. \\
        Many people act not out of conviction, but because they mistakenly believe that others are doing nothing.
    \end{alertblock}

\end{frame}

\begin{frame}{Backup: Key Problem of the RCT}
    Randomised Controlled Trial (RCT): Random assignment to treatment and control groups to identify causal effects through randomisation.

    \medskip
    
    \begin{itemize}
    \setlength{\itemsep}{8pt}
        \item Spillover effects
        \item John Henry effects (control group tries to keep up)
        \item \textbf{Hawthorne effects \& Social Desirability Bias: }Participants might report an inﬂated tendency to behave sustainably, e.g. due to a desire to appear good, which may result in an artiﬁcial gap between beliefs and behaviours/norms
        \item \textbf{Experimenter demand effects: }obfuscating the purpose of the information provision and incentivising the donation outcome
    \end{itemize}
    
\end{frame}
